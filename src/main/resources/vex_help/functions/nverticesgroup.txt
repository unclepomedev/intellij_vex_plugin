#type: vex
#context: all

#group: geo

"""Returns the number of vertices in the group."""

:usage: `int nverticesgroup(<geometry>, string groupname)`

:include _common#geometry:

:arg:groupname:
    A group name or ad-hoc group, like `0-3` or `@Cd.x>0.5`. This matches the
    SOP group naming convention, in particular that an empty string means all
    vertices.
